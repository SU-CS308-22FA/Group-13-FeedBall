  ageFunc(birth: any){
    var numtimestamp = Number(birth.seconds);
    numtimestamp = numtimestamp * 1000;
    const dateOf = new Date(numtimestamp);
    var birthyear = Number(dateOf.getFullYear());
    var time = new Date().getFullYear();
    var age = time - birthyear ;
    return age;
  }
