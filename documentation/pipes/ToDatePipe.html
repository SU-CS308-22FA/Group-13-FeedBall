@Pipe({ name: 'todatepipe' })
export class ToDatePipe implements PipeTransform {
transform(timestamp: any) {

  var numtimestamp = Number(timestamp.seconds);
  numtimestamp = numtimestamp * 1000;
  const dateOf = new Date(numtimestamp);

  return dateOf;
  }
}
